94AckermannTrajectoryControl::SteeringCommand AckermannTrajectoryControl::CurvatureToSteeringCommand(
95 const CurvatureCommand& command) const {
96 SteeringCommand steering_command;
97 steering_command.steering_angle = std::atan(command.kappa * wheelbase_);
98 if (!std::isfinite(command.kappa) || !std::isfinite(command.kappa_rate)) {
99 return steering_command;
100 }
101 const double cos_steering_angle = std::cos(steering_command.steering_angle);
102 steering_command.steering_angle_rate = command.kappa_rate * wheelbase_ * cos_steering_angle * cos_steering_angle;
103 return steering_command;
104}

107void AckermannTrajectoryControl::declareAndLoadParameter(const std::string& name,
108 T& param,
109 const std::string& description,
110 const bool add_to_auto_reconfigurable_params,
111 const bool is_required,
112 const bool read_only,
113 const std::optional<double>& from_value,
114 const std::optional<double>& to_value,
115 const std::optional<double>& step_value,
116 const std::string& additional_constraints) {
117 rcl_interfaces::msg::ParameterDescriptor param_desc;
118 param_desc.description = description;
119 param_desc.additional_constraints = additional_constraints;
120 param_desc.read_only = read_only;
121
122 auto type = rclcpp::ParameterValue(param).get_type();
123
124 if (from_value.has_value() && to_value.has_value()) {
125 if constexpr (std::is_integral_v<T>) {
126 rcl_interfaces::msg::IntegerRange range;
127 range.set__from_value(static_cast<T>(from_value.value())).set__to_value(static_cast<T>(to_value.value()));
128 if (step_value.has_value()) range.set__step(static_cast<T>(step_value.value()));
129 param_desc.integer_range = {range};
130 } else if constexpr (std::is_floating_point_v<T>) {
131 rcl_interfaces::msg::FloatingPointRange range;
132 range.set__from_value(static_cast<T>(from_value.value())).set__to_value(static_cast<T>(to_value.value()));
133 if (step_value.has_value()) range.set__step(static_cast<T>(step_value.value()));
134 param_desc.floating_point_range = {range};
135 } else {
136 RCLCPP_WARN(this->get_logger(), "Parameter type of parameter '%s' does not support specifying a range", name.c_str());
137 }
138 }
139
140 this->declare_parameter(name, type, param_desc);
141
142 try {
143 param = this->get_parameter(name).get_value<T>();
144 std::stringstream ss;
145 ss << "Loaded parameter '" << name << "': ";
146 if constexpr (is_vector_v<T>) {
147 ss << "[";
148 for (const auto& element : param) ss << element << (&element != &param.back() ? ", " : "");
149 ss << "]";
150 } else {
151 ss << param;
152 }
153 RCLCPP_INFO_STREAM(this->get_logger(), ss.str());
154 } catch (rclcpp::exceptions::ParameterUninitializedException&) {
155 if (is_required) {
156 RCLCPP_FATAL_STREAM(this->get_logger(), "Missing required parameter '" << name << "', exiting");

159 std::stringstream ss;
160 ss << "Missing parameter '" << name << "', using default value: ";

165 } else {
166 ss << param;
167 }
168 RCLCPP_WARN_STREAM(this->get_logger(), ss.str());
169 this->set_parameters({rclcpp::Parameter(name, rclcpp::ParameterValue(param))});
170 }
171 }
172
173 if (add_to_auto_reconfigurable_params) {
174 std::function<void(const rclcpp::Parameter&)> setter = [&param](const rclcpp::Parameter& p) { param = p.get_value<T>(); };
175 auto_reconfigurable_params_.push_back(std::make_tuple(name, setter));
176 }

179rcl_interfaces::msg::SetParametersResult AckermannTrajectoryControl::parametersCallback(
180 const std::vector<rclcpp::Parameter>& parameters) {
181 std::lock_guard<std::mutex> control_lock(control_mutex_);
182
183 for (const auto& param : parameters) {
184 for (auto& auto_reconfigurable_param : auto_reconfigurable_params_) {
185 if (param.get_name() == std::get<0>(auto_reconfigurable_param)) {
186 const bool old_use_odom = use_odom_;
187 std::get<1>(auto_reconfigurable_param)(param);
188 RCLCPP_INFO(this->get_logger(), "Reconfigured parameter '%s' to: %s", param.get_name().c_str(),
189 param.value_to_string().c_str());
190 if (param.get_name() == "use_odom" && use_odom_ != old_use_odom) {
191 ResetOdometry();
192 }
193 break;
194 }
195 }
196 if (param.get_name() == "max_curvature") {
197 if (!use_speed_dependent_lateral_limits_) {
198 max_curvature_current_ = max_curvature_;
199 }
200 }
201 if (param.get_name() == "max_curvature_rate") {
202 if (!use_speed_dependent_lateral_limits_) {
203 max_curvature_rate_current_ = max_curvature_rate_;
204 }
205 }
206 }
207
208 // mark parameter change successful
209 rcl_interfaces::msg::SetParametersResult result;
210 result.successful = true;
211
212 return result;
213}

215void AckermannTrajectoryControl::setup() {
216 // initialize dv-PID
217 dv_pid_ = std::make_unique<PID>(0.0, 0.0, 0.0);
218
219 // initialize dy-PID
220 dy_pid_ = std::make_unique<PID>(0.0, 0.0, 0.0);
221
222 // initialize dpsi-PID
223 dpsi_pid_ = std::make_unique<PID>(0.0, 0.0, 0.0);
224
225 // set Initial output values
226 vhcl_ctrl_output_.drive.steering_angle = 0.0;
227 vhcl_ctrl_output_.drive.steering_angle_velocity = 0.0;
228 vhcl_ctrl_output_.drive.speed = 0.0;
229 vhcl_ctrl_output_.drive.acceleration = 0.0;
230 vhcl_ctrl_output_.drive.jerk = 0.0;
231 vhcl_ctrl_output_.header.stamp = this->now();
232
233 ResetOdometry();
234
235 // initialize tf listener
236 tf2_buffer_ = std::make_unique<tf2_ros::Buffer>(this->get_clock());
237 tf2_listener_ = std::make_shared<tf2_ros::TransformListener>(*tf2_buffer_);
238
239 control_callback_group_ = create_callback_group(rclcpp::CallbackGroupType::MutuallyExclusive);
240
241 // initialize subscribers
242 vehicle_state_sub_ = create_subscription<perception_msgs::msg::EgoData>(
243 "~/ego_data", 1, std::bind(&AckermannTrajectoryControl::VehicleStateCallback, this, std::placeholders::_1));
244 trajectory_sub_ = create_subscription<trajectory_planning_msgs::msg::Trajectory>(
245 "~/trajectory", 1, std::bind(&AckermannTrajectoryControl::TrajectoryCallback, this, std::placeholders::_1));
246 lat_active_sub_ = create_subscription<std_msgs::msg::Bool>(
247 "~/lat_control_active", 1, std::bind(&AckermannTrajectoryControl::LatActiveCallback, this, std::placeholders::_1));
248 lon_active_sub_ = create_subscription<std_msgs::msg::Bool>(
249 "~/lon_control_active", 1, std::bind(&AckermannTrajectoryControl::LonActiveCallback, this, std::placeholders::_1));
250
251 // initialize publishers
252 vehicle_ctrl_pub_ = create_publisher<ackermann_msgs::msg::AckermannDriveStamped>("~/controls", 1);
253
254 if (use_speed_dependent_lateral_limits_) {
255 lateral_limits_loaded_ = LoadLateralLimitsCsv();
256 if (!lateral_limits_loaded_) {
257 RCLCPP_WARN_STREAM(get_logger(), "Speed dependent lateral limits disabled due to CSV load failure.");
258 }
259 }
260
261 // Initialize active curvature limits from the configured static or speed-dependent constraints.
262 UpdateLateralLimitsFromVelocity(0.0);
263
264 // initialize the cyclic vehicle-control timer; the callback VehicleCtrlCycle will be called wrt. the defined control frequency
265 last_cycle_time_ = now();
266 vhcl_ctrl_timer_ = create_wall_timer(std::chrono::duration<double>(1.0 / control_frequency_),
267 std::bind(&AckermannTrajectoryControl::VehicleCtrlCycle, this), control_callback_group_);
268
269 parameters_callback_ = this->add_on_set_parameters_callback(
270 std::bind(&AckermannTrajectoryControl::parametersCallback, this, std::placeholders::_1));
271
272 // Annotate message links for tracing: Publish ackermann commands periodically, depending an all subscribed topics.
273 std::vector<const void*> link_subs;
274 link_subs.push_back(static_cast<const void*>(vehicle_state_sub_->get_subscription_handle().get()));
275 link_subs.push_back(static_cast<const void*>(trajectory_sub_->get_subscription_handle().get()));
276 std::vector<const void*> link_pubs;
277 link_pubs.push_back(static_cast<const void*>(vehicle_ctrl_pub_->get_publisher_handle().get()));
278 TRACETOOLS_TRACEPOINT(message_link_periodic_async, link_subs.data(), link_subs.size(), link_pubs.data(), link_pubs.size());
279}

326void AckermannTrajectoryControl::setControllerGains() {
327 double velocity = perception_msgs::object_access::getVelLon(cur_vehicle_state_);
328 // feed-forward gains
329 if (!LinearInterpolation(gain_scheduling_velocity_lookup_, vec_feed_forward_gain_acceleration_, velocity,
330 feed_forward_gain_acceleration_)) {
331 feed_forward_gain_acceleration_ = 0.0;
332 }
333 if (!LinearInterpolation(gain_scheduling_velocity_lookup_, vec_feed_forward_gain_steering_angle_, velocity,
334 feed_forward_gain_steering_angle_)) {
335 feed_forward_gain_steering_angle_ = 0.0;
336 }
337
338 double p = 0.0, i = 0.0, d = 0.0;
339 // dv Controller
340 if (!LinearInterpolation(gain_scheduling_velocity_lookup_, dv_p_, velocity, p)) p = 0.0;
341 if (!LinearInterpolation(gain_scheduling_velocity_lookup_, dv_i_, velocity, i)) i = 0.0;
342 if (!LinearInterpolation(gain_scheduling_velocity_lookup_, dv_d_, velocity, d)) d = 0.0;
343 dv_pid_->SetParameters(p, i, d);
344
345 // dy Controller
346 if (!LinearInterpolation(gain_scheduling_velocity_lookup_, dy_p_, velocity, p)) p = 0.0;
347 if (!LinearInterpolation(gain_scheduling_velocity_lookup_, dy_i_, velocity, i)) i = 0.0;
348 if (!LinearInterpolation(gain_scheduling_velocity_lookup_, dy_d_, velocity, d)) d = 0.0;
349 dy_pid_->SetParameters(p, i, d);
350
351 // dpsi Controller
352 if (!LinearInterpolation(gain_scheduling_velocity_lookup_, dpsi_p_, velocity, p)) p = 0.0;
353 if (!LinearInterpolation(gain_scheduling_velocity_lookup_, dpsi_i_, velocity, i)) i = 0.0;
354 if (!LinearInterpolation(gain_scheduling_velocity_lookup_, dpsi_d_, velocity, d)) d = 0.0;
355 dpsi_pid_->SetParameters(p, i, d);
356}

358void AckermannTrajectoryControl::VehicleCtrlCycle() {
359 perception_msgs::msg::EgoData latest_vehicle_state;
360 trajectory_planning_msgs::msg::Trajectory latest_subscribed_trajectory;
361 bool latest_lat_active = true;
362 bool latest_lon_active = true;
363 uint64_t latest_trajectory_sequence = 0;
364
365 {
366 std::lock_guard<std::mutex> input_lock(input_mutex_);
367 latest_vehicle_state = latest_vehicle_state_;
368 latest_subscribed_trajectory = latest_subscribed_trajectory_;
369 latest_lat_active = latest_lat_active_;
370 latest_lon_active = latest_lon_active_;
371 latest_trajectory_sequence = latest_trajectory_sequence_;
372 }
373
374 std::lock_guard<std::mutex> control_lock(control_mutex_);
375 ctrl_time_ = now();
376 if (last_cycle_time_ > ctrl_time_) {
377 RCLCPP_WARN_STREAM(get_logger(), "Resetting controller because of Jump-Back in time!");
378 ResetController();
379 vhcl_ctrl_output_.header.stamp = ctrl_time_;
380 }
381
382 const double cycle_dt = (ctrl_time_ - last_cycle_time_).seconds();
383 if (cycle_dt > 1.25 / control_frequency_) {
384 RCLCPP_WARN_STREAM(get_logger(), "Exceeding the configured cycle period! dt since last timer cycle: "
385 << std::fixed << std::setprecision(15) << cycle_dt << " seconds.");
386 }
387 last_cycle_time_ = ctrl_time_;
388
389 cur_vehicle_state_ = latest_vehicle_state;
390
391 if (latest_trajectory_sequence != processed_trajectory_sequence_) {
392 processed_trajectory_sequence_ = latest_trajectory_sequence;
393 // check needs to be performed before any transformation because x=0, y=0, theta=0 is indicating a high-level-initialization
394 if (trajectory_planning_msgs::trajectory_access::getSamplePointSize(latest_subscribed_trajectory) > 0) {
395 // get x, y and theta of trajectory at first state
396 double x = trajectory_planning_msgs::trajectory_access::getX(latest_subscribed_trajectory, 0);
397 double y = trajectory_planning_msgs::trajectory_access::getY(latest_subscribed_trajectory, 0);
398 double theta = trajectory_planning_msgs::trajectory_access::getTheta(latest_subscribed_trajectory, 0);
399 if (x == 0.0 && y == 0.0 && theta == 0.0) { // high-level-initialization
400 dy_pid_->Reset();
401 dpsi_pid_->Reset();
402 dv_pid_->Reset();
403 }
404 }
405 }
406
407 if (VehicleStateOk(ctrl_time_)) {
408 // transform latest trajectory to current vehicle-frame
409 try {
410 tf_trajectory_ = tf2_buffer_->transform(latest_subscribed_trajectory, vehicle_frame_id_,
411 tf2_ros::fromMsg(cur_vehicle_state_.header.stamp), fixed_over_time_frame_id_,
412 tf2::durationFromSec(0.01));
413 } catch (tf2::TransformException& ex) {
414 RCLCPP_WARN(this->get_logger(), "Failed transforming trajectory in control cycle. Ex: %s", ex.what());
415 tf_trajectory_ = latest_subscribed_trajectory;
416 }
417 ResetOdometry();
418 UpdateLateralLimitsFromVelocity(perception_msgs::object_access::getVelLon(cur_vehicle_state_));
419 }
420 if (!latest_lat_active) {
421 dy_pid_->Reset();
422 dpsi_pid_->Reset();
423 }
424 if (!latest_lon_active) {
425 dv_pid_->Reset();
426 }
427 if (!InputSanityCheck()) // some inputs are not ok
428 {
429 RCLCPP_ERROR_STREAM(get_logger(), "Input sanity check failed! Skipping control cycle...");
430 return;
431 }
432
433 if (!TrjDataProc(ctrl_time_)) {
434 RCLCPP_ERROR_STREAM(get_logger(), "Processing of input Trajectory failed! Skipping control cycle...");
435 return;
436 }
437
438 double dt = (ctrl_time_ - vhcl_ctrl_output_.header.stamp).seconds();
439 if (dt <= 0.0) {
440 RCLCPP_ERROR_STREAM(get_logger(), "dt since last control output: " << std::fixed << std::setprecision(15) << dt
441 << " seconds. Skipping control cycle...");
442 return;
443 } else if (dt > 1.25 / control_frequency_) {
444 RCLCPP_WARN_STREAM(get_logger(), "Exceeding the expected dt since last control output! dt since last control output: "
445 << std::fixed << std::setprecision(15) << dt << " seconds.");
446 }
447 // Hold zero output (except delta) while the trajectory explicitly requests standstill.
448 if (tf_trajectory_.standstill) {
449 RCLCPP_DEBUG_STREAM(get_logger(), "Standstill.");
450 CurvatureCommand curvature_command{std::tan(delta_tgt_) / wheelbase_, 0.0};
451 LimitKappa(dt, curvature_command.kappa, curvature_command.kappa_rate, max_curvature_current_, max_curvature_rate_current_,
452 max_curvature_accel_, last_kappa_, last_kappa_rate_);
453 const SteeringCommand steering_command = CurvatureToSteeringCommand(curvature_command);
454 vhcl_ctrl_output_.drive.steering_angle = static_cast<float>(steering_command.steering_angle);
455 vhcl_ctrl_output_.drive.steering_angle_velocity = 0.0;
456 vhcl_ctrl_output_.drive.speed = 0.0;
457 double standstill_request_acceleration =
458 standstill_request_acceleration_gain_ * std::fabs(perception_msgs::object_access::getVelLon(cur_vehicle_state_));
459 standstill_request_acceleration = std::min(standstill_request_acceleration, 0.0);
460 standstill_request_acceleration = std::max(standstill_request_acceleration, lon_min_acc_);
461 vhcl_ctrl_output_.drive.acceleration = static_cast<float>(standstill_request_acceleration);
462 vhcl_ctrl_output_.drive.jerk = 0.0;
463 dy_pid_->Reset();
464 dpsi_pid_->Reset();
465 dv_pid_->Reset();
466 last_kappa_ = curvature_command.kappa;
467 last_kappa_rate_ = curvature_command.kappa_rate;
468 } else {
469 setControllerGains();
470 if (latest_lat_active) {
471 const CurvatureCommand curvature_command = LateralControl(dt);
472 const SteeringCommand steering_command = CurvatureToSteeringCommand(curvature_command);
473 if (std::isnan(steering_command.steering_angle)) {
474 RCLCPP_ERROR_STREAM(get_logger(), "Steering Angle Output Value isNaN! Skipping control cycle...");
475 vhcl_ctrl_output_.drive.steering_angle = 0.0;
476 vhcl_ctrl_output_.drive.steering_angle_velocity = 0.0;
477 dy_pid_->Reset();
478 dpsi_pid_->Reset();
479 return;
480 }
481 vhcl_ctrl_output_.drive.steering_angle = static_cast<float>(steering_command.steering_angle);
482 vhcl_ctrl_output_.drive.steering_angle_velocity = static_cast<float>(steering_command.steering_angle_rate);
483 } else {
484 const SteeringCommand steering_command =
485 UpdateKappaFromState(cur_vehicle_state_, wheelbase_, last_kappa_, last_kappa_rate_);

539bool AckermannTrajectoryControl::TrjDataProc(const rclcpp::Time& ctrl_time) {
540 // Derive State Vectors
541 std::vector<double> TIME, V, A, Y, THETA, DELTA;
542 int n_samples = trajectory_planning_msgs::trajectory_access::getSamplePointSize(tf_trajectory_);
543 for (int i = 0; i < n_samples; i++) {
544 TIME.push_back(trajectory_planning_msgs::trajectory_access::getT(tf_trajectory_, i));
545 V.push_back(trajectory_planning_msgs::trajectory_access::getV(tf_trajectory_, i));
546 Y.push_back(trajectory_planning_msgs::trajectory_access::getY(tf_trajectory_, i));
547 if (tf_trajectory_.type_id == trajectory_planning_msgs::DRIVABLE::TYPE_ID) {
548 A.push_back(trajectory_planning_msgs::trajectory_access::getA(tf_trajectory_, i));
549 THETA.push_back(trajectory_planning_msgs::trajectory_access::getTheta(tf_trajectory_, i));
550 DELTA.push_back(trajectory_planning_msgs::trajectory_access::getDeltaAck(tf_trajectory_, i));
551 } else {
552 RCLCPP_ERROR_STREAM(get_logger(),
553 "Unsupported trajectory type. Only trajectory_planning_msgs::DRIVABLE is currently supported.");
554 return false;
555 }
556 }
557
558 // calculate desired interpolation time for longitudinal values
559 double delta_time = (ctrl_time - tf_trajectory_.header.stamp).seconds();
560 // interpolate longitudinal target values
561 if (!LinearInterpolation(TIME, V, delta_time + lon_t_lookahead_, v_tgt_)) return false;
562 if (!LinearInterpolation(TIME, A, delta_time + lon_t_lookahead_, a_tgt_)) return false;
563
564 // interpolate lateral target values
565 if (!LinearInterpolation(TIME, Y, delta_time + lat_t_lookahead_, y_tgt_)) return false;
566 if (!LinearInterpolation(TIME, THETA, delta_time + lat_t_lookahead_, psi_tgt_)) return false;
567 if (!LinearInterpolation(TIME, DELTA, delta_time + lat_t_lookahead_, delta_tgt_)) return false;
568
569 if (use_odom_) {
570 // CalcOdometry
571 double dt_ego = (ctrl_time - cur_vehicle_state_.header.stamp).seconds();
572 double dt_ctrl = (ctrl_time - vhcl_ctrl_output_.header.stamp).seconds();
573 double dt = std::min(dt_ego, dt_ctrl);
574 CalcOdometry(dt); // Cyclic Control
575
576 dy_ = odom_dy_ - y_tgt_;
577 dpsi_ = odom_dpsi_ - psi_tgt_;
578 } else {
579 dy_ = -y_tgt_;
580 dpsi_ = -psi_tgt_;
581 }
582 return true;
583}

585bool AckermannTrajectoryControl::LinearInterpolation(const std::vector<double>& X,
586 const std::vector<double>& Y,
587 const double& desired_x,
588 double& output_y) {
589 if (desired_x < *min_element(X.begin(), X.end()) || desired_x > *max_element(X.begin(), X.end())) {
590 RCLCPP_ERROR_STREAM(get_logger(), "Desired x value is outside the range covered by the input vector.");
591 return false;
592 }
593 if (X.size() != Y.size()) {
594 RCLCPP_ERROR_STREAM(get_logger(), "Input vectors don't have the same length!");
595 return false;
596 }
597
598 // go through array and search for sampling points
599 size_t i = 0;
600 for (i = 0; i < X.size(); i++) {
601 if (X[i] < desired_x) {
602 continue;
603 } else if (X[i] == desired_x) {
604 output_y = Y[i];
605 return true;
606 } else {
607 break;
608 }
609 }
610 output_y = Y[i - 1] + ((Y[i] - Y[i - 1]) / (X[i] - X[i - 1])) * (desired_x - X[i - 1]);
611 return true;
612}

767bool AckermannTrajectoryControl::LimitKappa(const double dt,
768 double& kappa_tgt,
769 double& kappa_rate,
770 const double max_curvature,
771 const double max_curvature_rate,
772 const double max_curvature_accel,
773 const double kappa_prev,
774 const double kappa_rate_prev) {
775 if (dt <= 0.0) {
776 RCLCPP_ERROR_STREAM(get_logger(), "Non-positive dt. Skipping curvature limiting...");
777 return false;
778 }
779
780 bool kappa_limited = false;
781 if (fabs(kappa_tgt) > max_curvature) {
782 if (kappa_tgt >= 0.0) {
783 kappa_tgt = max_curvature;
784 } else {
785 kappa_tgt = -max_curvature;
786 }
787 dy_pid_->ResetIntegral();
788 RCLCPP_WARN_STREAM(get_logger(), "Curvature limited!");
789 kappa_limited = true;
790 }
791
792 kappa_rate = 0.0;
793 kappa_rate = (kappa_tgt - kappa_prev) / dt;
794
795 if (fabs(kappa_rate) > max_curvature_rate && dt > 0.0) {
796 if (kappa_rate >= 0.0) {
797 kappa_rate = max_curvature_rate;
798 } else {
799 kappa_rate = -max_curvature_rate;
800 }
801 kappa_tgt = kappa_prev + kappa_rate * dt;
802 dy_pid_->ResetIntegral();
803 RCLCPP_WARN_STREAM(get_logger(), "Curvature-rate limited!");
804 kappa_limited = true;
805 }
806
807 double kappa_accel = 0.0;
808 kappa_accel = (kappa_rate - kappa_rate_prev) / dt;
809 if (fabs(kappa_accel) > max_curvature_accel && dt > 0.0) {
810 if (kappa_accel >= 0.0) {
811 kappa_accel = max_curvature_accel;
812 } else {
813 kappa_accel = -max_curvature_accel;
814 }
815 kappa_rate = kappa_rate_prev + kappa_accel * dt;
816 kappa_tgt = kappa_prev + kappa_rate * dt;
817 dy_pid_->ResetIntegral();
818 RCLCPP_WARN_STREAM(get_logger(), "Curvature-acceleration limited!");
819 kappa_limited = true;
820 }
821
822 return kappa_limited;
823}

825AckermannTrajectoryControl::CurvatureCommand AckermannTrajectoryControl::LateralControl(const double dt) {
826 const bool vehicle_standstill = perception_msgs::object_access::getStandstill(cur_vehicle_state_);
827 if (vehicle_standstill) {
828 // we reset the PID controllers in standstill to avoid integral windup and undesired overshoot when starting from standstill
829 dy_pid_->ResetIntegral();
830 dpsi_pid_->ResetIntegral();
831 }
832
833 // cascaded control
834 double w_y = 0.0;
835 double e_y = w_y - dy_;
836 double w_psi = dy_pid_->Calc(e_y, dt);
837 double e_psi = w_psi - dpsi_;
838 double psi_dot_des = dpsi_pid_->Calc(e_psi, dt);
839
840 double velocity = perception_msgs::object_access::getVelLon(cur_vehicle_state_);
841 // be sure v!=0 (to avoid division by zero)
842 if (fabs(velocity) < 0.1) {
843 if (velocity < 0.0) {
844 velocity = -0.1;
845 } else {
846 velocity = 0.1;
847 }
848 }
849
850 double kappa_pid = std::tan(psi_dot_des * (wheelbase_ + self_st_gradient_ * velocity * velocity) / velocity) / wheelbase_;
851
852 // ackermann feed-forward control (convert delta to kappa for feed-forward)
853 double kappa_ff = std::tan(delta_tgt_) / wheelbase_;
854
855 double kappa_tgt = kappa_pid + kappa_ff * feed_forward_gain_steering_angle_;
856
857 double kappa_rate = 0.0;
858 bool kappa_is_limited = LimitKappa(dt, kappa_tgt, kappa_rate, max_curvature_current_, max_curvature_rate_current_,
859 max_curvature_accel_, last_kappa_, last_kappa_rate_);
860
861 if (kappa_is_limited && !vehicle_standstill) {
862 if (use_back_calculation_) {
863 double kappa_fb_sat = kappa_tgt - kappa_ff * feed_forward_gain_steering_angle_;
864 double denom = (wheelbase_ + self_st_gradient_ * velocity * velocity) / velocity;
865 double psi_dot_sat = std::atan(kappa_fb_sat * wheelbase_) / denom;
866 dpsi_pid_->BackCalculate(psi_dot_des, psi_dot_sat, dt, anti_windup_gain_);
867 } else {
868 dpsi_pid_->ResetIntegral();
869 }
870 }
871
872 last_kappa_ = kappa_tgt;
873 last_kappa_rate_ = kappa_rate;
874 return CurvatureCommand{kappa_tgt, kappa_rate};
875}

877AckermannTrajectoryControl::LongitudinalCommand AckermannTrajectoryControl::LongitudinalControl(const double dt) {
878 LongitudinalCommand longitudinal_command;
879 double velocity = perception_msgs::object_access::getVelLon(cur_vehicle_state_);
880 double w_v = v_tgt_;
881 double e_v = w_v - velocity;
882 double a_fb_v = dv_pid_->Calc(e_v, dt);
883
884 double a_ff = a_tgt_ * feed_forward_gain_acceleration_;
885 double a_ctrl = a_fb_v + a_ff;
886 double a_unsat = a_ctrl;
887
888 // limit desired acceleration
889 if (a_ctrl > lon_max_acc_) {
890 a_ctrl = lon_max_acc_;
891 RCLCPP_WARN_STREAM(get_logger(), "Longitudinal acceleration limited!");
892 } else if (a_ctrl < lon_min_acc_) {
893 a_ctrl = lon_min_acc_;
894 RCLCPP_WARN_STREAM(get_logger(), "Longitudinal acceleration limited!");
895 }
896
897 // calculate jerk with respect to last desired acceleration
898 double jerk = 0.0;
899 if (dt > 0.0) {
900 jerk = (a_ctrl - vhcl_ctrl_output_.drive.acceleration) / dt;
901 }
902 if (fabs(jerk) > lon_max_jerk_ && dt > 0.0) {
903 if (jerk >= 0.0) {
904 jerk = lon_max_jerk_;
905 } else {
906 jerk = -lon_max_jerk_;
907 }
908 a_ctrl = vhcl_ctrl_output_.drive.acceleration + jerk * dt;
909 RCLCPP_WARN_STREAM(get_logger(), "Longitudinal jerk limited!");
910 }
911 if (a_ctrl != a_unsat) {
912 if (use_back_calculation_) {
913 double a_fb_sat = a_ctrl - a_ff;
914 dv_pid_->BackCalculate(a_fb_v, a_fb_sat, dt, anti_windup_gain_);
915 } else {
916 dv_pid_->ResetIntegral();
917 }
918 }
919
920 longitudinal_command.speed = v_tgt_;
921 longitudinal_command.acceleration = a_ctrl;
922 longitudinal_command.jerk = jerk;
923 return longitudinal_command;
924}
